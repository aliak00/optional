    Enum member isOptional

    Checks if T is an optional type

    enum isOptional(T) = isInstanceOf!(Optional, T);

    Example

    import optional: Optional;

assert(isOptional!(Optional!int) == true);
assert(isOptional!int == false);
assert(isOptional!(int[]) == false);
